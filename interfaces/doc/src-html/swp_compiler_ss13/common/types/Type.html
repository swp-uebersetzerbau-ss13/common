001package swp_compiler_ss13.common.types;
002
003/**
004 * Base Type for type definitions.
005 * 
006 * @author "Frank Zechert", "Danny Maasch"
007 * @version 1
008 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/wiki/Types">Types Wiki</a>
009 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/issues/8">Types Issue Tracker</a>
010 */
011public abstract class Type {
012        
013        /**
014         * Width of this type in bytes.
015         */
016        protected Long width;
017
018        /**
019         * Returns the width of this type in bytes.
020         * @return Number of bytes for this type.
021         */
022        public long getWidth() {
023                return this.width;
024        }
025        
026        /**
027         * Get the name of the type.
028         * @return The name of the type.
029         */
030        public abstract String getTypeName();
031        
032        /**
033         * Returns a nice string representation of this type
034         * @return String representation of this type
035         */
036        public abstract String toString();
037}
