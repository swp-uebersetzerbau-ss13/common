001package swp_compiler_ss13.common.optimization;
002
003/**
004 * Proposed interface for liveliness information
005 * @author "Frank Zechert", "Danny Maasch"
006 * @version 1
007 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/wiki/Symbol Table">Symbol Table Wiki</a>
008 * @see <a target="_top" href="https://github.com/swp-uebersetzerbau-ss13/common/issues/6">Symbol Table Issue Tracker</a>
009 */
010public interface Liveliness {
011        
012        /**
013         * Set whether the identifier is alive or not
014         * @param lives true to set the identifier to be alive
015         */
016        public void setLiveliness(Boolean lives);
017        
018        /**
019         * Checks whether the identifier is alive or not
020         * @return true if the identifier is alive.
021         */
022        public Boolean isAlive();
023        
024        /**
025         * Set the line this identifier is used in next.
026         * @param line The line this identifier is used in next. null for none.
027         */
028        public void setNextUse(Integer line);
029        
030        /**
031         * Gets the next line this identifier is used in next.
032         * @return The line this identifier is used in next or null for none.
033         */
034        public Integer getNextUse();
035}
