006/**
007 * Exception thrown when intermediate code generation fails.
008 * 
009 * @author "Frank Zechert", "Danny Maasch"
010 * @version 1
011 * @see <a target="_top" href=
012 *      "https://github.com/swp-uebersetzerbau-ss13/common/wiki/Intermediate-Code-Generator"
013 *      >Intermediate Code Generator Wiki</a>
014 * @see <a target="_top"
015 *      href="https://github.com/swp-uebersetzerbau-ss13/common/issues/1"
016 *      >Intermediate Code Generator Issue Tracker</a>
017 */
018public class IntermediateCodeGeneratorException extends Exception {
019
020        /**
021         * serial version uid
022         */
023        private static final long serialVersionUID = -8670103678115061426L;
024
025        /**
026         * Create a new IntermediateCodeGeneratorException.
027         */
028        public IntermediateCodeGeneratorException() {
029        }
030
031        /**
032         * Create a new IntermediateCodeGeneratorException.
033         * 
034         * @param message
035         *            The error message.
036         */
037        public IntermediateCodeGeneratorException(String message) {
038                super(message);
039        }
040
041        /**
042         * Create a new IntermediateCodeGeneratorException.
043         * 
044         * @param cause
045         *            The error cause.
046         */
047        public IntermediateCodeGeneratorException(Throwable cause) {
048                super(cause);
049        }
050
051        /**
052         * Create a new IntermediateCodeGeneratorException.
053         * 
054         * @param message
055         *            The error message.
056         * @param cause
057         *            The error cause.
058         */
059        public IntermediateCodeGeneratorException(String message, Throwable cause) {
060                super(message, cause);
061        }
062}
